add_subdirectory(${CMAKE_CURRENT_SOURCE_DIR}/SimpleUDP)

# GLFW+OpenGL https://github.com/glfw/glfw.git
# requires sudo apt install libwayland-dev libxkbcommon-dev wayland-protocols extra-cmake-modules libwayland-bin libxrand-dev libxinerama-dev libxcursor-dev libxi-dev                    
find_package(OpenGL REQUIRED)
set(GLFW_BUILD_DOCS OFF CACHE BOOL "" FORCE)
set(GLFW_BUILD_TESTS OFF CACHE BOOL "" FORCE)
set(GLFW_BUILD_EXAMPLES OFF CACHE BOOL "" FORCE)
add_subdirectory(${CMAKE_CURRENT_SOURCE_DIR}/glfw)

# Dear ImGui https://github.com/ocornut/imgui
set(IMGUI_DIR ${CMAKE_CURRENT_SOURCE_DIR}/imgui)
file(GLOB IMGUI_SOURCES ${IMGUI_DIR}/*.cpp)
add_library(imgui STATIC ${IMGUI_SOURCES} ${IMGUI_DIR}/backends/imgui_impl_glfw.cpp
                                          ${IMGUI_DIR}/backends/imgui_impl_opengl3.cpp)
target_include_directories(imgui PUBLIC ${IMGUI_DIR} ${IMGUI_DIR}/backends)
target_link_libraries(imgui PUBLIC OpenGL::GL glfw dl)

# Mavlink2 autogenerated headers https://github.com/mavlink/c_library_v2.git
# no source files
add_subdirectory(${CMAKE_CURRENT_SOURCE_DIR}/mavlink)
target_include_directories(mavlink INTERFACE $<BUILD_INTERFACE:${CMAKE_CURRENT_BINARY_DIR}/mavlink/include>)
add_compile_options(-Wno-address-of-packed-member)
add_subdirectory(${CMAKE_CURRENT_SOURCE_DIR}/googletest)
